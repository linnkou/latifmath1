// focus-game.js function checkFocusAnswer() { const answer = document.getElementById('focus-answer').value.toLowerCase(); const result = document.getElementById('focus-result'); if (answer === 'تركيز') { result.textContent = 'إجابة صحيحة!'; result.style.color = 'green'; } else { result.textContent = 'إجابة خاطئة، حاول مرة أخرى.'; result.style.color = 'red'; } } // إعداد المؤقت let timeLeft = 30; const timerElement = document.getElementById('timer'); function updateTimer() { timerElement.textContent = timeLeft; if (timeLeft > 0) { timeLeft--; setTimeout(updateTimer, 1000); } else { alert('انتهى الوقت!'); } } updateTimer();
